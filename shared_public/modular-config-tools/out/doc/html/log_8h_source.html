119 #define CTUTIL_LOG_INTERNAL_STATUS(enable, logLevel, status, message) \
120  if(enable) \
121  { \
122  char const * szMessage = message; \
123  char szMessageBuffer[MAX_MESSAGE_BUFFER_SIZE]; \
124  if(status != 0) \
125  { \
126  int const formatResult = snprintf(szMessageBuffer, MAX_MESSAGE_BUFFER_SIZE, "%s (%s %i)", message, "status code", status); \
127  if(formatResult < 0) \
128  { \
129  ctutil_Log(ctutil_LogLevelEnum_warning, NULL, __FILE__, __func__, __LINE__, "Format error"); \
130  } \
131  else if(formatResult >= MAX_MESSAGE_BUFFER_SIZE) \
132  { \
133  ctutil_Log(ctutil_LogLevelEnum_warning, NULL, __FILE__, __func__, __LINE__, "Message text too long to add status code"); \
134  } \
135  else \
136  { \
137  szMessage = szMessageBuffer; \
138  } \
139  } \
140  ctutil_Log(logLevel, NULL, __FILE__, __func__, __LINE__, szMessage); \
141  }
